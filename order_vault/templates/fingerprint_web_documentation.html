Rediim Fingerprint – Quick Start

Create a stable per‑device ID (visitorId) and a per‑session ID (localSessionId) on the web.

What it does

Rediim Fingerprint creates:

  • visitorId – a stable, per‑device identifier
  • localSessionId – a per‑tab/visit session id

Call it right after login (or on any page you want to track). Pass your own user_id so we can analyze duplicates across users/devices. Also you can pass the promocode used or offered so we can analyze abuse across users/devices.

1) Install: dynamic import

No bundler or NPM required. Dynamically import our script and call a single function.

<script>
// Optional: Generate an Access token
async function getAccessTokenClient() { // Client side API
  try {
    const res = await fetch("https://order-vault-client-webapp-13ee822f0ba9.herokuapp.com/rediim/token", {
      method: "POST",
      credentials: "include"
    });

    if (!res.ok) {
      console.warn("Failed to get access token:", res.status, res.statusText);
      return null; // return null instead of throwing
    }

    const data = await res.json();
    return data.access_token || null;
  } catch (err) {
    console.error("Error fetching access token:", err);
    return null;
  }
}
    
  // Optional: your own user identifier. If you don’t have one, use 0 or skip.
  function getUserId() {
    let uid = localStorage.getItem("user_id");
    if (!uid) {
      uid = Math.floor(Math.random() * 1_000_000);
      localStorage.setItem("user_id", uid);
    }
    return uid;
  }

  // Call this on pages you want to track (commonly right after login)
  async function runInHouseFingerprint(key, clientId, call_type = null, userId = null, coupon = null) {
    try {
      const InHouseFingerprint = await import('https://api.rediim.com/static/js/fingerprint_web.js');
      const { visitorId, localSessionId } = await InHouseFingerprint.sendFingerprint(key, clientId, call_type, userId, coupon);

      // Optional: Cache for reuse across the site
      localStorage.setItem("rediim_fingerprint", visitorId);
      localStorage.setItem("local_session_id", localSessionId);

      // Optional: expose globally if you like
      window.REDIIM = { visitorId, localSessionId };

      console.log("[Rediim] Fingerprint ready", { visitorId, localSessionId });
      return { visitorId, localSessionId };
    } catch (error) {
      console.error("[Rediim] Error getting fingerprint:", error);
      // Don’t block UX; you can retry later
      return null;
    }
  }

  // Example: run on page load (post-login is ideal)
  window.addEventListener('load', async () => {
    
   // Store safely Publishable Key (env injected or server templated)
    const REDIIM_PK_KEY = "pk_key";   // replace
    const REDIIM_CLIENT_ID = "client_id";  // replace
    
    // Or generate an Access Token from your backened
    const accessToken = await getAccessTokenClient();
    
    const userId = getUserId(); // optional but recommended
    const coupon = {promocode: promocode, // optional but recommended
                    promotion_id: promotion_id,
                    type: type,
            }; 
    const call_type = "Login";
    const { visitorId, localSessionId } = await runInHouseFingerprint(accessToken, REDIIM_CLIENT_ID, call_type, userId, coupon);
  });
</script>

Node.js (Express)
import express from "express";
import jwt from "jsonwebtoken";

const app = express();
const CLIENT_ID = process.env.CLIENT_ID;
const SHARED_SECRET = process.env.SHARED_SECRET;

app.post("/rediim/token", (req, res) => {
  const now = Math.floor(Date.now() / 1000);
  const payload = {
    cid: CLIENT_ID,
    scope: ["fingerprint"],
    aud: "rediim-api",
    iat: now,
    exp: now + 15 * 60,
  };
  const token = jwt.sign(payload, SHARED_SECRET, { algorithm: "HS256" });
  res.json({ access_token: token });
});

app.listen(3000);

Go (net/http)
package main

import (
	"encoding/json"
	"net/http"
	"os"
	"time"

	"github.com/golang-jwt/jwt/v5"
)

var clientID = os.Getenv("CLIENT_ID")
var shared = []byte(os.Getenv("SHARED_SECRET"))

func rediimToken(w http.ResponseWriter, r *http.Request) {
	now := time.Now()
	claims := jwt.MapClaims{
		"cid":   clientID,
		"scope": []string{"fingerprint"},
		"aud":   "rediim-api",
		"iat":   now.Unix(),
		"exp":   now.Add(15 * time.Minute).Unix(),
	}
	t := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
	signed, _ := t.SignedString(shared)

	w.Header().Set("Content-Type", "application/json")
	json.NewEncoder(w).Encode(map[string]string{"access_token": signed})
}

func main() {
	http.HandleFunc("/rediim/token", rediimToken)
	http.ListenAndServe(":3000", nil)
}

Java (Spring Boot)
@RestController
public class TokenController {
  @Value("${CLIENT_ID}") String clientId;
  @Value("${SHARED_SECRET}") String shared;

  @PostMapping("/rediim/token")
  public Map token() {
    long now = System.currentTimeMillis() / 1000;
    String jwt = io.jsonwebtoken.Jwts.builder()
      .claim("cid", clientId)
      .claim("scope", List.of("fingerprint"))
      .audience().add("rediim-api").and()
      .issuedAt(new java.util.Date(now * 1000))
      .expiration(new java.util.Date((now + 15*60) * 1000))
      .signWith(io.jsonwebtoken.security.Keys.hmacShaKeyFor(shared.getBytes()),
                io.jsonwebtoken.SignatureAlgorithm.HS256)
      .compact();
    return Map.of("access_token", jwt);
  }
}

C# (ASP.NET Core Minimal API)
using System.IdentityModel.Tokens.Jwt;
using Microsoft.IdentityModel.Tokens;
using System.Security.Claims;

var builder = WebApplication.CreateBuilder(args);
var app = builder.Build();

var clientId = Environment.GetEnvironmentVariable("CLIENT_ID");
var secret = Environment.GetEnvironmentVariable("SHARED_SECRET");
var key = new SymmetricSecurityKey(System.Text.Encoding.UTF8.GetBytes(secret));

app.MapPost("/rediim/token", () => {
    var now = DateTimeOffset.UtcNow;
    var creds = new SigningCredentials(key, SecurityAlgorithms.HmacSha256);
    var claims = new List {
        new("cid", clientId),
        new("aud", "rediim-api"),
        // scope array: put as JSON string or multiple "scope" claims; here JSON:
        new("scope", "[\"fingerprint\"]")
    };
    var token = new JwtSecurityToken(
        claims: claims,
        notBefore: now.UtcDateTime,
        expires: now.AddMinutes(15).UtcDateTime,
        signingCredentials: creds
    );
    var jwt = new JwtSecurityTokenHandler().WriteToken(token);
    return Results.Json(new { access_token = jwt });
});

app.Run();

2) Returned values & storage

Values

  • visitorId – stable device id
  • localSessionId – per‑visit/per‑tab session id

Where we store them

  • localStorage["rediim_fingerprint"]
  • localStorage["local_session_id"]

This enables duplicate detection and device reuse analysis for promo abuse prevention.

3) Common pitfalls

TypeError: Cannot set properties of null
You’re writing into an element that isn’t in the DOM yet. Use DOMContentLoaded/load or guard with if (el). If you don’t use hidden inputs, remove those lines.

CSP / Adblock / Mixed Content
Serve over HTTPS and allow https://api.rediim.com in script-src and connect-src.

Key/Client mismatch
X-API-KEY and X-CLIENT-ID must belong together. If you rotate keys, update both.

Retries
sendFingerprint() is idempotent. If it fails (network/CSP), don’t block UX—log and retry later.

4) Monitoring your traffic in the Rediim Fingerprint Dashboard

Once you’ve integrated the fingerprint snippet on your site, you can monitor device and session activity in the Rediim Fingerprint Dashboard.

  • See real-time visitorId and localSessionId creation events.
  • Search by user_id to find all devices used by the same account.
  • Search by device_id (visitorId) to see how many accounts use the same device.
  • Filter by promocode to detect promo abuse patterns.

To get meaningful insights:

  • Always pass a user_id to sendFingerprint() when known.
  • If applicable, pass the promocode offered or used at the time of fingerprinting.
  • Ensure your frontend sends the same device_id and local_session_id to your backend for risk evaluation calls.

With these identifiers in place, the dashboard will let you monitor traffic, detect duplicate devices across users, and catch suspicious promo usage.
